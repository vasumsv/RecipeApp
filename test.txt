protected Provision[] mapListToArray(List<Provision> provisionList) {
    if (provisionList == null) {
        return new Provision[0]; // Correct: Directly return an empty Provision array
    }
    Provision[] provisionArray = new Provision[provisionList.size()];
    int i = 0;
    for (Provision provision : provisionList) {
        provisionArray[i++] = provision;
    }
    return provisionArray;
}
